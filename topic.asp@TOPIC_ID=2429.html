разобрался. сделал через ф-ю SetViewRect

Dim sh As Visio.Shape

Set sh = ActivePage.Shapes("Sheet.1")
Application.ActiveWindow.DeselectAll
Application.ActiveWindow.Select sh, visSelect
Application.ActiveWindow.SetViewRect _
sh.Cells("PinX") - 2, sh.Cells("PinY") + 2, 4, 4
