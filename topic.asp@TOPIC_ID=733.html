Public Sub DrawRect()
Const InchToMM = 1 / 25.4
Dim x1, x2, y1, y2 As Double
x1 = 10 * InchToMM
y1 = 10 * InchToMM
x2 = 30 * InchToMM
y2 = 30 * InchToMM
Call ActivePage.DrawRectangle(x1, y1, x2, y2)
End Sub
