Таки ж она рассказывает, что not set.
Значит надо Set
Примерно так:
Sub ttt()
Dim X() As Visio.Shape
ReDim X(1)
Set X(1) = ActivePage.Shapes(1)
ReDim Preserve X(2)
Set X(2) = ActivePage.Shapes(2)
Debug.Print X(1).NameID
Debug.Print X(2).NameID
End Sub
