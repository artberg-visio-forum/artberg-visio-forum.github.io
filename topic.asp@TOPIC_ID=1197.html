Наверное можно попробовать в обработчике события открытия документа поискать это окно и включить, если выключено
Типа вот так:

Private Sub Document_DocumentOpened(ByVal doc As IVDocument)
    For i = 1 To Application.Windows(1).Windows.Count
        If InStr(1, Application.Windows(1).Windows(i).Caption, "Custom Properties") = 1 Then
            If Not Application.Windows(1).Windows(i).Visible Then
                Application.Windows(1).Windows(i).Visible = True
            End If
        End If
    Next
End Sub
